C program
#include <stdio.h>

int main() {
    for (int x = 1; x <= 10000; x++) {

        int sum = 0; 
        for (int i = 1; i < x; i++) {
            if (x % i == 0) {
                sum += i;
            }
        }
        
        if (sum == x) {
            printf("%d\n", x);
        }
    }

    return 0;
}
